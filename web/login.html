async function login() { const user = document.getElementById("username").value; const pass = document.getElementById("password").value; const res = await fetch("/login", { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ username: user, password: pass }) }); const data = await res.json(); if (data.ok) { window.location.href = "index.html"; } else { document.getElementById("msg").innerHTML = data.error || "Invalid username or password"; } }
